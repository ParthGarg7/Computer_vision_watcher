# The Watcher — Computer Vision Pipeline
# Dependencies (Layers 1-5)
#
# Install all with:
#   pip install -r requirements.txt
#
# PyTorch must be installed separately with the correct CUDA index URL.
# For RTX 5060 (Blackwell sm_120) with CUDA 13.2 driver:
#   pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu128
#
# The yolov8n-face.pt model is NOT installed via pip.
# Download it with:
#   python -c "from huggingface_hub import hf_hub_download; import shutil; shutil.copy(hf_hub_download('arnabdhar/YOLOv8-Face-Detection', 'model.pt'), 'models/yolov8n-face.pt')"
#
# IMPORTANT: opencv-python-headless conflict.
# insightface may pull in opencv-python-headless as a sub-dependency.
# If both are installed, headless wins and disables all GUI (imshow etc).
# Fix: pip uninstall opencv-python-headless -y && pip install --force-reinstall opencv-python

# ─── Core computer vision ─────────────────────────────────────────────────────
# IMPORTANT: Use opencv-python (NOT opencv-python-headless).
opencv-python>=4.13.0

# ─── Numerical computing ──────────────────────────────────────────────────────
numpy>=2.0.0

# ─── YOLOv8 face detection (Layer 3) ─────────────────────────────────────────
ultralytics>=8.4.0

# ─── Model download utility (used to fetch yolov8n-face.pt from HuggingFace) ──
huggingface_hub>=1.0.0

# ─── PyTorch — install manually with the correct CUDA URL (see above) ─────────
# torch>=2.7.0
# torchvision>=0.22.0
# torchaudio>=2.7.0

# ─── Layer 4: Identity ────────────────────────────────────────────────────────
# InsightFace: ArcFace face embedding + SCRFD alignment (buffalo_l model pack)
# First run downloads ~500 MB buffalo_l weights to ~/.insightface (one-time).
insightface>=0.7.3

# DeepSORT: Multi-face temporal tracking with ArcFace appearance swap.
# Uses the ArcFace 512-d embedding as appearance descriptor (pedestrian model swapped out).
deep-sort-realtime>=1.3.2

# FAISS: Fast nearest-neighbour search for identity registry (IndexFlatIP for MVP).
# Use faiss-gpu for GPU-accelerated search in production.
faiss-cpu>=1.8.0

# ─── Layer 5: Expression Analysis ────────────────────────────────────────────
# hsemotion-onnx: AffectNet-trained 8-class emotion classifier (ONNX Runtime).
# Chosen over DeepFace because Python 3.14 has no TensorFlow wheels.
# First run downloads ~50 MB ONNX model weights (one-time).
hsemotion-onnx

# ONNX Runtime: inference backend for hsemotion-onnx AND InsightFace.
# For GPU inference (recommended — ~4ms vs ~50-80ms per expression crop):
#   pip uninstall onnxruntime -y && pip install onnxruntime-gpu
# CUDA DLLs are provided by the PyTorch install; src/core/gpu_setup.py
# registers them on the Windows DLL search path automatically.
onnxruntime>=1.18.0

# ─── Layer 8: REST API + Layer 9 dashboard server ────────────────────────────
# FastAPI serves the read API over stored sessions; uvicorn runs it.
# Start with: python run_api.py  →  http://127.0.0.1:8000
fastapi>=0.115.0
uvicorn[standard]>=0.30.0
